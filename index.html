        const yesBtn = document.getElementById('yesBtn');

        const noBtn = document.getElementById('noBtn');

        const container = document.getElementById('quizContainer');

        const question = document.getElementById('question');

        const subtext = document.getElementById('subtext');

        const btnGroup = document.getElementById('btnGroup');


        // Funkcija koja pomiče "NE" gumb na nasumičnu poziciju unutar ekrana

        function bjeziGumbu() {

            // Izračunaj maksimalne granice ekrana uzimajući u obzir veličinu gumba

            const maxX = window.innerWidth - noBtn.offsetWidth - 20;

            const maxY = window.innerHeight - noBtn.offsetHeight - 20;


            // Generiraj nasumične koordinate

            const randomX = Math.floor(Math.random() * maxX);

            const randomY = Math.floor(Math.random() * maxY);


            // Prebaci gumb u fixed poziciju kako bi mogao bježati po cijelom ekranu

            noBtn.style.position = 'fixed';

            noBtn.style.left = randomX + 'px';

            noBtn.style.top = randomY + 'px';

        }


        // Bježi na hover (miš na PC-u)

        noBtn.addEventListener('mouseenter', bjeziGumbu);

        

        // Bježi na dodir (mobitel)

        noBtn.addEventListener('touchstart', function(e) {

            e.preventDefault(); // Sprječava "klikanje" na mobitelu prije nego što pobjegne

            bjeziGumbu();

        });


        // Što se dogodi kada klikne DA

        yesBtn.addEventListener('click', () => {

            // Promijeni tekst i izgled

            question.innerHTML = “Ehh, ovom se nisam nad’o hahahah”;

            subtext.innerHTML = “Ne moramo se žurit onda, vidimo se vcrs/sutra</strong>";

            

            // Sakrij gumbe

            btnGroup.style.display = 'none';

            document.body.style.background = "linear-gradient(135deg, #ff9999, #ff4d4d)";

        });
